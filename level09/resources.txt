#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>

int main()
{
    const char *token_path = "/home/user/level09/token"; // Ruta absoluta al token
    FILE *file = fopen(token_path, "rb"); // Modo binario para leer bytes crudos
    
    if (!file)
    {
        fprintf(stderr, "Error: No se pudo abrir el archivo %s\n", token_path);
        return 1;
    }

    printf("token descifrada: ");
    int i = 0;
    int c;
    while ((c = fgetc(file)) != EOF)
	{ // Lee byte a byte
        	char decrypted = c - i++;
        	if (isprint(decrypted)) 
		{     // Filtra caracteres imprimibles
            	putchar(decrypted);
        }
    }
    fclose(file);
    printf("\n");
    return 0;
}
